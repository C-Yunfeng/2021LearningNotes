      ```sql
      select 学生姓名，表a.学生学号，学科成绩
      from 表a,表b
      where 学科='数学'
      ORDER BY 学科成绩 DESC 
      limit 3;

      from 表a

      and 学科='数学'
      ORDER BY 学科成绩 DESC
      limit 3;
      ```
